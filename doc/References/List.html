def initialize(*vals)
	if vals.length == 0
		@head = nil
		@last = nil
	else
		for val in vals do
			pushEnd val
		end
	end
end

def length

end

def put(val)
	if @head == nil
		@head = @last = Node.new(val,nil,nil)
	else
		aux = Node.new(val,@head, nil)
		@head.back = aux
		@head = aux
	end
	nil
end
