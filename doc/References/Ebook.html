def formatAPA
  (prettyOutput(@authors.map { |x| x.to_s }) + "(" + @datee.year.to_s + ") " + @title +
   "\n\t(" + @edition.to_s + ") " +
   "(" + @editionnumber.to_s + ") " +
   @url)
end
